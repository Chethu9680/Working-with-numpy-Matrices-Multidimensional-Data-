# Import necessary libraries
import numpy as np
import matplotlib.pyplot as plt
import pandas as pd
from scipy.stats import zscore, pearsonr, spearmanr
import seaborn as sns

# 1. Download and load the data
url_male = 'https://github.com/gagolews/teaching-data/raw/master/marek/nhanes_adult_male_bmx_2020.csv'
url_female = 'https://github.com/gagolews/teaching-data/raw/master/marek/nhanes_adult_female_bmx_2020.csv'

male = np.genfromtxt(url_male, delimiter=',', skip_header=1)
female = np.genfromtxt(url_female, delimiter=',', skip_header=1)

# 2. Load the data as numpy matrices
print("Male Matrix:")
print(male[:5])  # Print first 5 rows
print("Female Matrix:")
print(female[:5])  # Print first 5 rows

# 3. Plot histograms for male and female weights
plt.figure(figsize=(10, 6))

plt.subplot(2, 1, 1)
plt.hist(female[:, 0], bins=20, color='pink', edgecolor='black')
plt.title('Histogram of Female Weights')
plt.xlim(30, 140)

plt.subplot(2, 1, 2)
plt.hist(male[:, 0], bins=20, color='blue', edgecolor='black')
plt.title('Histogram of Male Weights')
plt.xlim(30, 140)

plt.tight_layout()
plt.show()

# 4. Draw box-and-whisker plots to compare male and female weights
plt.figure(figsize=(8, 6))
plt.boxplot([female[:, 0], male[:, 0]], labels=['Female', 'Male'])
plt.title('Boxplot of Female and Male Weights')
plt.show()

# 5. Compute basic numerical aggregates and describe the distributions
female_weights_stats = pd.Series(female[:, 0]).describe()
male_weights_stats = pd.Series(male[:, 0]).describe()

print("Female Weights Summary Statistics:")
print(female_weights_stats)
print("\nMale Weights Summary Statistics:")
print(male_weights_stats)

# 6. Add BMI column to the female matrix
female_bmi = female[:, 0] / (female[:, 1]/100) ** 2  # BMI = weight (kg) / height (m)^2
female = np.column_stack((female, female_bmi))

# 7. Standardize the female dataset
zfemale = zscore(female, axis=0)

# 8. Draw scatterplot matrix (pairplot) for standardised variables
df_zfemale = pd.DataFrame(zfemale, columns=['Weight', 'Height', 'Arm Length', 'Leg Length', 
                                            'Arm Circumference', 'Hip Circumference', 
                                            'Waist Circumference', 'BMI'])
sns.pairplot(df_zfemale[['Weight', 'Height', 'Waist Circumference', 'Hip Circumference', 'BMI']])
plt.show()

# Compute and interpret Pearsons and Spearmans correlation coefficients
corr_pearson = df_zfemale.corr(method='pearson')
corr_spearman = df_zfemale.corr(method='spearman')

print("Pearson Correlation Coefficients:\n", corr_pearson)
print("\nSpearman Correlation Coefficients:\n", corr_spearman)

# 9. Compute waist-to-height and waist-to-hip ratios for male and female participants
male_waist_to_height = male[:, 6] / male[:, 1]
female_waist_to_height = female[:, 6] / female[:, 1]

male_waist_to_hip = male[:, 6] / male[:, 5]
female_waist_to_hip = female[:, 6] / female[:, 5]

male = np.column_stack((male, male_waist_to_height, male_waist_to_hip))
female = np.column_stack((female, female_waist_to_height, female_waist_to_hip))

# 10. Draw box-and-whisker plots for the waist-to-height and waist-to-hip ratios
plt.figure(figsize=(10, 6))
plt.boxplot([female[:, -2], male[:, -2], female[:, -1], male[:, -1]], 
            labels=['Female Waist-to-Height', 'Male Waist-to-Height', 
                    'Female Waist-to-Hip', 'Male Waist-to-Hip'])
plt.title('Comparison of Waist-to-Height and Waist-to-Hip Ratios')
plt.show()

# 11. Advantages and Disadvantages of BMI, Waist-to-Height, and Waist-to-Hip Ratios
advantages_disadvantages = {
    "BMI": {
        "Advantages": ["Easy to calculate.", "Widely used and accepted."],
        "Disadvantages": ["Does not distinguish between muscle and fat.", "May not be accurate for athletes."]
    },
    "Waist-to-Height Ratio": {
        "Advantages": ["Better indicator of fat distribution.", "Simple to calculate."],
        "Disadvantages": ["Less known and used than BMI."]
    },
    "Waist-to-Hip Ratio": {
        "Advantages": ["Good indicator of abdominal fat.", "Can predict cardiovascular risk."],
        "Disadvantages": ["Not as simple to interpret.", "Less widely recognized."]
    }
}

print("Advantages and Disadvantages of BMI, Waist-to-Height, and Waist-to-Hip Ratios:")
print(advantages_disadvantages)

# 12. Standardized body measurements for participants with the lowest and highest BMI
sorted_indices = np.argsort(zfemale[:, -1])
lowest_bmi = zfemale[sorted_indices[:5]]
highest_bmi = zfemale[sorted_indices[-5:]]

print("Standardized Measurements for 5 Persons with Lowest BMI:\n", lowest_bmi)
print("\nStandardized Measurements for 5 Persons with Highest BMI:\n", highest_bmi)
